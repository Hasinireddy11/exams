**PIPELINE CODE**

pipeline {
    agent any
    tools{
        maven 'MAVEN_HOME'
    }
    stages {
        stage('git repo & clean') {
            steps {
                //bat "rmdir  /s /q mavenjava"
                bat "git clone https://github.com/srivijju/jenkins_web.git"
                bat "mvn clean -f jenkins_web"
            }
        }
        stage('install') {
            steps {
                bat "mvn install -f jenkins_web"
            }
        }
        stage('test') {
            steps {
                bat "mvn test -f jenkins_web"
            }
        }
        stage('package') {
            steps {
                bat "mvn package -f jenkins_web"
            }
        }
    }
}

pipeline {
    agent any

    tools {
        maven 'maven'   // Jenkins Maven tool name from Global Tool Configuration
    }

    stages {
        stage('Clean Workspace') {
            steps {
                bat 'rmdir /s /q maven-web 2>nul || exit 0'
            }
        }

        stage('Clone Repository') {
            steps {
                bat 'git clone https://github.com/Hasinireddy11/maven-web.git'
            }
        }

        stage('Maven Clean') {
            steps {
                bat 'mvn clean -f maven-web/pom.xml'
            }
        }

        stage('Maven Install') {
            steps {
                bat 'mvn install -f maven-web/pom.xml'
            }
        }

        stage('Maven Test') {
            steps {
                bat 'mvn test -f maven-web/pom.xml'
            }
        }

        stage('Maven Package') {
            steps {
                bat 'mvn package -f maven-web/pom.xml'
            }
        }
    }
}
